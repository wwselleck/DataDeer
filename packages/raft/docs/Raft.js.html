/**
 * Main raft class
 * @static
 * @public
 */
class Raft {
  /**
   * @param {module:lib/raft~RaftConfig} config
   * @public
   */
  constructor (config, store) {
    this.config = config
    this.store = store

    const { dataSources } = config
    this._applySources(dataSources)
  }

  /**
   * @param sources
   * @instance
   */
  _applySources (sources) {
    log.debug({sources}, 'Attempting to apply sources')
    Object.keys(sources).forEach(sourceName => {
      const source = sources[sourceName]
      this.use(sourceName, source)
    })
  }

  /**
   * Get active sources
   * @return {Object.<string, lib/raft.SourceConfig>}
   * @memberof lib/raft.Raft
   * @instance
   * @public
   */
  sources () {
    return this.store.getSources()
  }

  /**
   * Use a source
   * @param {string} id
   * @param {lib/raft.SourceConfig} sourceConfig
   * @memberof lib/raft.Raft
   * @instance
   * @public
   */
  use (id, sourceConfig) {
    log.debug({sourceConfig}, 'Attemping to apply source')
    this.store.addSource(id, sourceConfig)
    log.debug({sourceConfig}, 'Source applied')
    return this
  }

  /**
   * Get a source
   * @param {string} id - Id of source to get
   * @returns {lib/raftDataSource.RaftDataSource}
   * @memberof lib/raft.Raft
   * @instance
   */
  get (id) {
    return this.store.get(id)
  }

  /**
   * Get all default data from all sources
   * @returns {Object}
   * @memberof lib/raft.Raft
   * @instance
   */
  fetch () {
    log.debug('::fetch')
    return this.store.fetch()
  }
}

module.exports = {
  /**
   * Creates an instance of Raft
   * @param {Raft.RaftConfig} config
   * @returns {Raft.Raft}
   * @memberof lib/raft
   * @public
   */
  create (config) {
    return new Raft(config, RaftDataStore.create())
  },
  OptionTypes
}
